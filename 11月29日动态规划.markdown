#### 198.打家劫舍

```c++
class Solution {
public:
    int rob(vector<int>& nums) {
        if(nums.size() == 0) return 0;
        if(nums.size() == 1) return nums[0];
        vector<int> dp(nums.size(), 0);
        dp[0] = nums[0];
        dp[1] = max(nums[0], nums[1]);
        for(int i = 2; i < nums.size(); ++i)
            dp[i] = max(dp[i - 1], dp[i - 2] + nums[i]);
        return dp[nums.size() - 1];
    }
};
```

此题就是对i位置的家来说，如果我们选择打劫，那么我们就从i-2的位置得到的累积结果去添加。如果我们不选择打劫，那么我们不添加，保持前一家的结果，注意初始化。

#### 213.打家劫舍II

```c++
class Solution {
public:
    int rob(vector<int>& nums) {
        if(nums.size() == 1) return nums[0];
        if(nums.size() == 2) return max(nums[0], nums[1]);
        //因为首元素和尾元素必然不可能同时选取，那么可以完全等价成以下两种情况
        int result1 = rob_range(nums, 0, nums.size() - 2);
        int result2 = rob_range(nums, 1, nums.size() - 1);
        return max(result1, result2);
    }

    int rob_range(vector<int>& nums, int start, int end){
        int eNums = start - end + 1;
        if(eNums == 1) return nums[start];
        if(eNums == 2) return max(nums[start], nums[start + 1]);
        vector<int> dp(nums.size(), 0);
        dp[start] = nums[start];
        dp[start + 1] = max(nums[start], nums[start + 1]);
        for(int i = start + 2; i <= end; ++i){
            dp[i] = max(dp[i - 1], dp[i - 2] + nums[i]);
        }
        return dp[end];
    }
};
```

这道题的思路就是把环拆开，变成两种情况，去掉首元素或者去掉尾元素。然后注意一下dp数组最好是和nums对齐，然后让两者都通过start end来操作。

#### 337.打家劫舍III

```c++
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    vector<int> robTree(TreeNode* node){
        vector<int> dp(2, 0);
        if(node == nullptr) return dp;
        vector<int> left = robTree(node->left);
        vector<int> right = robTree(node->right);
        //pos 0表示不偷当前值，pos 1表示偷当前值
        dp[0] = max(left[0], left[1]) + max(right[0], right[1]);
        dp[1] = node->val + left[0] + right[0];
        return dp;
    }
    int rob(TreeNode* root) {
        vector<int> result = robTree(root);
        return max(result[0], result[1]);
    }
};
```

这道题是树形dp，dp只有两个选项，拿/不拿当前节点，利用递归完成计算。
