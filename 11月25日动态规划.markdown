#### 卡码网46.携带研究材料

```c++
#include <iostream>
#include <vector>
using namespace std;

int main(){
    int m, n;
    cin >> n >> m;
    // m: 背包空间，n: 材料
    vector<int> weight(n, 0);
    vector<int> value(n, 0);
    for(int i = 0; i < n; ++i)
        cin >> weight[i];
    for(int i = 0; i < n; ++i)
        cin >> value[i];
    vector<vector<int>>dp(n, vector<int>(m + 1, 0));
    //每一行：一种材料
    //每一列：从小到大的一种空间取值
    //初始化
    for(int i = weight[0]; i <= m; ++i){
        dp[0][i] = value[0];
    }
    for(int i = 1; i < n; ++i){
        for(int j = 0; j <= m; ++j){
            if(j < weight[i]) dp[i][j] = dp[i - 1][j];
            else
                dp[i][j] = max(dp[i - 1][j], dp[i - 1][j - weight[i]] + value[i]);
        }
    }
    cout << dp[n - 1][m];
    return 0;
}
```

#### 416.分割等和子集

```c++
class Solution {
public:
    bool canPartition(vector<int>& nums) {
        int sum = 0;
        for(int i = 0; i < nums.size(); ++i)
            sum += nums[i];
        if(sum % 2 == 1) return false;
        sum /= 2;
        vector<int> dp(sum + 1, 0);
        for(int i = 0; i < nums.size(); ++i){
            for(int j = sum; j >= nums[i]; --j){
                dp[j] = max(dp[j], dp[j - nums[i]] + nums[i]);
            }
        }
        return dp[sum] == sum;
    }
};
```

这里我转换成了填满最大结果除以2的背包问题。具体来说，这里我们让weight和value一致，我们代码的目标是找到最大value，那在这种情况下一定是装满sum/2的空间，这也就表明存在这样的组合。所以算法是正确的。
